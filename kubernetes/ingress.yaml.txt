apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: web-app-ingress 
  namespace: web-app
  annotations:
    kubernetes.io/ingress.class: nginx 
    cert-manager.io/cluster-issuer: letsencrypt-prod
spec:
  tls:
  - hosts:
    - example.com
    secretName: tls-secrete
  rules:
  - host: example.com 
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: web-app-service 
            port:
              number: 80
